--nombre, apellido, salario y departamento del empleado
--cursor
--todos los empleados que TRABAJAN COMO PROGRAMMER
--bloque simple
--mostrar consola
SET SERVEROUTPUT ON;
DECLARE
CURSOR CUR_EMP(P_TRABAJO JOBS.JOB_TITLE%TYPE) IS 
    SELECT
        E.FIRST_NAME||' '||E.LAST_NAME AS NOMBRE_EMPLEADO,
        E.SALARY,
        D.DEPARTMENT_NAME,
        J.JOB_TITLE
        FROM EMPLOYEES E
        INNER JOIN DEPARTMENTS D ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
        INNER JOIN JOBS J ON E.JOB_ID = J.JOB_ID
        WHERE (JOB_TITLE=P_TRABAJO);
BEGIN
    FOR REC_EMP IN CUR_EMP('Programmer') LOOP
        DBMS_OUTPUT.PUT_LINE('EL NOMBRE DEL EMPLEADO ES: '||REC_EMP.NOMBRE_EMPLEADO);
        DBMS_OUTPUT.PUT_LINE('EL SUELDO DEL EMPLEADO ES: '||REC_EMP.SALARY);
        DBMS_OUTPUT.PUT_LINE('EL NOMBRE DEL DEPARTAMENTO DEL EMPLEADO ES: '||REC_EMP.DEPARTMENT_NAME);
        DBMS_OUTPUT.PUT_LINE('EL CARGO DEL EMPLEADO ES: '||REC_EMP.JOB_TITLE);
        DBMS_OUTPUT.PUT_LINE('--------------------------------------------------------');
    END LOOP;
END;
